class Array:
    def __init__(self):
        self.data = {}
        self.length = 0
    
    def push(self, item):
        self.data[self.length] = item
        self.length += 1
    
    def pop(self):
        if self.length == 0:
            return None
        self.length -= 1
        return self.data.pop(self.length)
    
    def get(self, index):
        return self.data.get(index)
    
    def delete(self, index):
        for i in range(index, self.length-1):
            self.data[i] = self.data[i+1]
        del self.data[self.length-1]
        self.length -= 1

# Example usage
arr = Array()
arr.push(10)
arr.push(20)
arr.push(30)
print(arr.get(1))  # Output: 20
arr.delete(1)
print(arr.get(1))  # Output: 30

class Stack:
    def __init__(self):
        self.items = []
    
    def push(self, item):
        self.items.append(item)
    
    def pop(self):
        if not self.is_empty():
            return self.items.pop()
        return None
    
    def peek(self):
        if not self.is_empty():
            return self.items[-1]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)

# Example usage
stack = Stack()
stack.push(10)
stack.push(20)
stack.push(30)
print(stack.pop())  # Output: 30
print(stack.peek()) # Output: 20

def infix_to_postfix(expression):
    precedence = {'+': 1, '-': 1, '*': 2, '/': 2, '^': 3}
    output = []
    stack = []
    
    for char in expression:
        if char.isalnum():
            output.append(char)
        elif char == '(':
            stack.append(char)
        elif char == ')':
            while stack and stack[-1] != '(':
                output.append(stack.pop())
            stack.pop()
        else:
            while (stack and stack[-1] != '(' and 
                   precedence.get(char, 0) <= precedence.get(stack[-1], 0)):
                output.append(stack.pop())
            stack.append(char)
    
    while stack:
        output.append(stack.pop())
    
    return ''.join(output)

# Example usage
infix = "a+b*(c^d-e)^(f+g*h)-i"
postfix = infix_to_postfix(infix)
print("Infix:", infix)
print("Postfix:", postfix)
