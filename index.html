class Array:
    def __init__(self):
        self.data = {}
        self.length = 0
    
    def push(self, item):
        self.data[self.length] = item
        self.length += 1
    
    def pop(self):
        if self.length == 0:
            return None
        self.length -= 1
        return self.data.pop(self.length)
    
    def get(self, index):
        return self.data.get(index)
    
    def delete(self, index):
        for i in range(index, self.length-1):
            self.data[i] = self.data[i+1]
        del self.data[self.length-1]
        self.length -= 1

class Stack:
    def __init__(self):
        self.items = []
    
    def push(self, item):
        self.items.append(item)
    
    def pop(self):
        if not self.is_empty():
            return self.items.pop()
        return None
    
    def peek(self):
        if not self.is_empty():
            return self.items[-1]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)
